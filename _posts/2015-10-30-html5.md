最近公司需要做一个手机版的报表，想想自己画一个说不定更好一点，所写的代码封装了一下，下面把这个东西记录了下来
使用的是 `html5` 下面的 `canvas` 画布，当然，代码还有待改进

目前想画的是三种基本的报表

> 1. 类似于股市的报表 ，我们这里取名为 `走势图`，数据量非常多的时候，方便显示
> 2. 柱形的报表，数据量较多的时候显示
> 3. 圆形报表

首先定义整体的报表实体
	
	// 保存传递过来的参数，用于实例化报表
	var _params;
	
	// 报表对象
	var Report = function (params) {
		
		// 临时保存传递过来的参数
		_params = params;
		
		// 用于画图的报表 canvas 画布，画图就在这个上面操作
		this.control = params.control;
		
		// 2d 画图对象
		this.ctx = params.control.getContext('2d');
		
		// 画布的宽度，这里画布只能用像素来定义，不能使用百分比
		this.cwidth = 300;
		
		// 画布高度
		this.cheight = 250;
		
		// 需要画的数据
		this.data = params.data;
	}

*使用临时变量 `_params` 保存当前传递过来的变量， 这么使用的原因是这样的， 这里一起画了3个报表类型 ， 我们需要根据不同的报表 实例化出不同的参数类型，避免浪费，所以先把参数临时保存下来，需要实例化的时候再取出来用，我们首先画型报表*

	// 初始化圆形报表参数
	Report.prototype.initCircle = function () {
	
		// 圆心所在的X坐标，坐标为 canvas 画布的宽度的一半 减掉40个像素
		// 为什么要减掉40个像素，是因为圆形报表上面还需要标注说明文字
		// 可是说明文字是按照给定的坐标从左到右开始显示的，这样就会显示到右边一点，
		// 所以需要把圆心向左边移动一点，经过测试 40像素差不多，当然也可以根据实际情况自己调
		this.circleCenterX = this.cwidth / 2 - 40;
		
		// 圆心所在的y坐标，为 canvas 画布高度的一半
		this.circleCenterY = this.cheight / 2;
		
		// 圆半径，默认为95像素
		this.circleRadius = _params.radius || 95;
		
		// 传递过来的总数，用于计算每一部分所占的百分比
		this.circleTotal = this.getTotal();
		
		// 每个数值所代表的度数， 总数为 2*Math.PI 
		this.circleUnit = 2 * Math.PI / _params.total;
		
		// 圆的每一块所代表的颜色，这里只列出了12项
		this.circleColor = ['#43A521', '#A6B528', '#CAD856', '#E4DB7A', '#ECDE48', '#ECBF48', '#EC9548', '#D97330', '#D95430', '#A80000', '#740000', '#530000'];
	}

	// 获得当前给定值的总条数，用于计算每一部分所占的百分比
	Report.prototype.getTotal = function () {
		var total = 0;
		for (var item in this.data) {
			total += this.data[item] * 1;
		}
		return total;
	}

如此圆的数据初始化就完成了，下面我们只需要按要求画出圆就算大功完成了，下面是实现的步骤

	// 画出圆形报表，每一次画图时，画出的图形是一个扇形，最后连成了一个圆
	Report.prototype.drawCircle = function () {
		
		// 清空画布
		this.clear();
		
		// 初始化圆形报表参数
		this.initCircle();
		
		// 初始化图形索引，从0开始 一个挨一个画
		this.circleIndex = 0;
		
		// 圆心开始的弧度，这里是从0开始
		var startAngle = 0;
		
		// 圆心结束的弧度
		var endAngle = 0;
		
		// 扇形的提示文字，提示文字应该在扇形的中间,
		var textAngle = 0;
		
		// 判断传递过来的参数是否有值，因为传递过来的是对象，所以不能判断length,只能循环判断了。
		// 目前还没有找到更好的办法。
		var isEmpty = true;
		
		// 循环遍历传递过来的数组，画出给定的图形
		for (var item in this.data) {
			
			// 计算结束的弧度
			// 结束的弧度 = 开始的弧度 - this.data里面传递过来的数值所占的弧度
			endAngle = startAngle - this.data[item] * this.circleUnit;
			
			// 文字所在的弧度，等于扇形所在弧度的一半
			// 这里特别要注意，不是 ()startPoint - this.data[item] * this.circleUnit)/2
			// 一开始就是卡在这里了的
			textAngle = startAngle - this.data[item] * this.circleUnit/2;
			
			// 给定所画扇形的颜色
			this.ctx.fillStyle = this.circleColor[this.circleIndex];
			
			// 设置扇形边框的颜色，如果不设置这个颜色，可能会出现白边
			this.ctx.strokeStyle = this.circleColor[this.circleIndex];
		
			// 开始画图
			this.ctx.beginPath();
			
			// 坐标点移动到圆心
			this.ctx.moveTo(this.circleCenterX, this.circleCenterY);
			
			// 画出扇形
			this.ctx.arc(this.circleCenterX, this.circleCenterY, this.circleRadius, startAngle, endAngle, true);
			
			// 设置扇形颜色
			this.ctx.fill();
			
			// 给扇形边框设置颜色，防止白边
			this.ctx.stroke();
			
			// 画扇形上面的提示字体，这里需要用到高中学的数学知识
			// 开始画图
			this.ctx.beginPath();
			
			// 设置扇形区域中央提示字体的样式
			this.ctx.font = "small-caps normal 16px Arial";
			
			// 设置字体的颜色为黑色
			this.ctx.fillStyle = "black";
			
			// 计算字体所在的坐标点
			// 根据圆心角和半径求 提示字体所在的坐标点
			// 提示字体所在的坐标点的 X = 圆心X坐标 + cos(圆心角) * 半径
		    // 提示字体所在的坐标点的 Y = 圆心Y坐标 + sin(圆心角) * 半径
			// 因为字体需要在圆内，我这里显示的是在圆心 2/3 的位置 所以半径需要再 乘以 2/3
			this.ctx.fillText(this.data[item], this.circleCenterX + Math.cos(textAngle) * this.circleRadius * 2 / 3, this.circleCenterY + Math.sin(textAngle) * this.circleRadius * 2 / 3);
			
			// 提示字体在圆的外面，所以得大于半径，考虑到手机的屏幕比较小， 所以半径再乘以 1.15 差不多
			this.ctx.fillText(item, this.circleCenterX + Math.cos(textAngle) * this.circleRadius * 1.15, this.circleCenterY + Math.sin(textAngle) * this.circleRadius * 1.15);
			
			// 下一个扇形的开始弧度等于当前扇形的结束弧度
			startAngle = endAngle;
			
			// 画下一个扇形的颜色索引
			this.circleIndex++;
			
			// 设置不为空，因为this.data里面有值 
			isEmpty = false;
			
			// 回调行数，把画图的参数传递给网页
			this.setRemind(item, this.data[item]);
		}
		
		// 判断数据是否为空
		if (isEmpty) {
			this.drawEmpty();
		}
	}

	// 初始化圆柱图参数
	Report.prototype.initReverseCircle = function () {
		// 图形X轴坐标
		this.coordX = _params.diceLeft || 25;
		// 图形Y轴坐标
		this.coordY = _params.diceTop || 30;
		// 图形X轴的宽度 = canvas的宽度 - X轴坐标宽度
		this.coordWidth = this.cwidth - this.coordX;
		// 图形Y轴的长度 = canvas的高度 - Y轴坐标高度
		this.coordHeight = this.cheight - this.coordY;
		// X轴单位，默认Y轴分成6等分，原因是手机屏幕不适合显示太多的东西
		this.unitX = _params.unitX || 6;
		// X轴的最大值
		this.maxX = _params.maxX || 30;
		// Y轴单位，默认分成5等分
		this.unitY = _params.unitY || 5;
		// Y轴最大值
		this.maxY = _params.maxY || 100;
		// X轴最小单位
		this.minUnitX = this.coordWidth / this.maxX;
		// Y轴最小单位
		this.minUnitY = this.coordHeight / this.maxY;
	}
