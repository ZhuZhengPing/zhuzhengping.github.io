## 概念

ExcelExport 是一个简单的 excel 导出类库，ExcelExport 使用 OleDb 生成 Excel 文件










### 简单示例

```c#
new ExcelExport()
    .AddSheet("Sample Names", new[] {
        new { Name = "Bill Stewart",    ZipCode = "00347", Birth_Date = new DateTime(1987, 6, 5) },
        new { Name = "Russ Porter",     ZipCode = "04257", Birth_Date = new DateTime(1956, 7, 8) },
        new { Name = "Rodrick Rivers",  ZipCode = "19867", Birth_Date = new DateTime(1956, 7, 8) }
    })
    .AddSheet(
        "LINQ Query Sample",
        ordersQuery.Select(o => new { 
            Product_Name = o.Product.Name, 
            o.Quantity,
            o.OrderDate
        }
    )
    .AddSheet(someDataSet.Tables[0])
    .AddSheet(
        "Classic ADO.Net Sample",
        someCommand.ExecuteReader()
    )
    .ExportTo(Path.Combine(
        Environment.GetFolderPath(Environment.SpecialFolder.MyDocuments), 
        "Sample.xlsx"
    ));
```

### 注意

>* 当导出类型为匿名类型时，匿名类型属性中的 _ 会被替换成空格
>* 当导出 ADO.NET DataTables时，sheet的名字是可选的，如果指定了，sheet的名字就是DateTable的名字
>* 当导出 ADO.Net DataReaders时，reader必须保持打开，当 ExportTo()方法执行时，reader 将被关闭

[点击此处查看源码地址](https://github.com/SLaks/ExcelExporter)



### 导入类

```c#
public class ExcelReportHelper
{
	/// <summary>
	/// 导出 Excel
	/// </summary>
	/// <typeparam name="T">导出集合实体</typeparam>
	/// <param name="collection">导出集合</param>
	/// <param name="excelName">文件名称</param>
	/// <returns>导出数据流</returns>
	public byte[] ExportOut<T>(IEnumerable<T> collection)
	{
		ExcelExport excel = new ExcelExport();

		excel.AddSheet("Sheet1", collection);

		string path = AppDomain.CurrentDomain.SetupInformation.ApplicationBase + "ExportOutput";
		if (!System.IO.Directory.Exists(path))
			System.IO.Directory.CreateDirectory(path);

		path += "\\" + Guid.NewGuid().ToString() + ".xlsx";

		excel.ExportTo(path);

		byte[] fileBytes = System.IO.File.ReadAllBytes(path);

		try
		{
			System.IO.File.Delete(path);
		}
		catch (Exception e)
		{
			//Log4netHelper.Error("删除文件：“" + path + "”出错：" + e.Message, e);
		}
		return fileBytes;
	}

	/// <summary>
	/// 导入
	/// </summary>
	/// <param name="ImportClass">数据流</param>
	/// <param name="action">回调函数</param>
	public string ExportIn(HttpPostedFileBase ImportClass,Func<DataTable,string> action)
	{
		int FileLen = ImportClass.ContentLength;
		byte[] FileData = new byte[FileLen];
		Stream sr = ImportClass.InputStream;//创建数据流对象 
		sr.Read(FileData, 0, FileLen);
		sr.Close();

		string path = AppDomain.CurrentDomain.SetupInformation.ApplicationBase + "ExportOutput";
		if (!System.IO.Directory.Exists(path))
			System.IO.Directory.CreateDirectory(path);

		path += "\\" + Guid.NewGuid().ToString() + ".xlsx";
		File.WriteAllBytes(path, FileData);

		string strConn = string.Format("Provider=Microsoft.ACE.OLEDB.12.0;Data Source={0};Extended Properties='Excel 8.0;HDR=Yes;IMEX=1;'", path);
		OleDbConnection conn = new OleDbConnection(strConn);
		conn.Open();
		string strExcel = "";
		OleDbDataAdapter myCommand = null;
		DataSet ds = null;
		strExcel = "select * from [sheet1$]";
		myCommand = new OleDbDataAdapter(strExcel, strConn);
		ds = new DataSet();
		myCommand.Fill(ds, "table1");

		conn.Close();
		myCommand = null;

		// 删除
		try
		{
			File.Delete(path);
		}
		catch
		{
		}

		return action(ds.Tables[0]);
	}
}
```
